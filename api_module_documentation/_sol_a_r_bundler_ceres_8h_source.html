146 inline void toRodrigues(datastructure::Transform3Df &R, datastructure::Vector3f&r) {
147 const double small = 1e-6;
148 double th = acos
149 (0.5*(fmax(R(0, 0) + R(1, 1) + R(2, 2), -1.0) - 1.0));
150 double sth = sin(th);
151 double cth = cos(th);
152 if (fabs(sth) < small && cth < 0) {
153 double W_pt[9], x, y, z;
154 W_pt[0] = 0.5*(R(0, 0) + R(0, 0)) - 1.0;
155 W_pt[1] = 0.5*(R(1, 0) + R(0, 1));
156 W_pt[2] = 0.5*(R(2, 0) + R(0, 2));
157
158 W_pt[3] = 0.5*(R(0, 1) + R(1, 0));
159 W_pt[4] = 0.5*(R(1, 1) + R(1, 1)) - 1.0;
160 W_pt[5] = 0.5*(R(2, 1) + R(1, 2));
161
162 W_pt[6] = 0.5*(R(0, 2) + R(2, 0));
163 W_pt[7] = 0.5*(R(1, 2) + R(2, 1));
164 W_pt[8] = 0.5*(R(2, 2) + R(2, 2)) - 1.0;
165
166
167 x = sqrt(0.5 * (W_pt[0] - W_pt[4] - W_pt[8]));
168 y = sqrt(0.5 * (W_pt[4] - W_pt[8] - W_pt[0]));
169 z = sqrt(0.5 * (W_pt[8] - W_pt[0] - W_pt[4]));
170
171
172 if (x >= y && x >= z) {
173 y = (W_pt[1] >= 0) ? y : -y;
174 z = (W_pt[2] >= 0) ? z : -z;
175 }
176 else if (y >= x && y >= z) {
177 z = (W_pt[5] >= 0) ? z : -z;
178 x = (W_pt[1] >= 0) ? x : -x;
179 }
180 else {
181 x = (W_pt[2] >= 0) ? x : -x;
182 y = (W_pt[5] >= 0) ? y : -y;
183 }
184
185 {
186 double scale = th / sqrt(1 - cth);
187 r(0) = scale * x;
188 r(1) = scale * y;
189 r(2) = scale * z;
190
191 return;
192 }

194 }

208 inline void iRodrigues(datastructure::Vector3d &r, datastructure::Transform3Df&R) {
209
210 const double small = 1e-6;
211
212 double th = sqrt(r(0)*r(0) +
213 r(1)*r(1) +
214 r(2)*r(2));
215
216 if (th < small) {
217 R(0, 0) = 1.0; R(0, 1) = 0.0; R(0, 2) = 0.0;
218 R(1, 0) = 0.0; R(1, 1) = 1.0; R(1, 2) = 0.0;
219 R(2, 0) = 0.0; R(2, 1) = 0.0; R(2, 2) = 1.0;
220 return;
221 }
222
223 {
224 double x = r(0) / th;
225 double y = r(1) / th;
226 double z = r(2) / th;
227
228 double xx = x * x;
229 double xy = x * y;
230 double xz = x * z;
231 double yy = y * y;
232 double yz = y * z;
233 double zz = z * z;
234
235 const double yx = xy;
236 const double zx = xz;
237 const double zy = yz;
238
239 double sth = sin(th);
240 double cth = cos(th);
241 double mcth = 1.0 - cth;
242
243 R(0, 0) = 1 - mcth * (yy + zz);
244 R(1, 0) = sth * z + mcth * xy;
245 R(2, 0) = -sth * y + mcth * xz;
246
247 R(0, 1) = -sth * z + mcth * yx;
248 R(1, 1) = 1 - mcth * (zz + xx);
249 R(2, 1) = sth * x + mcth * yz;
250
251 R(0, 2) = sth * y + mcth * xz;
252 R(1, 2) = -sth * x + mcth * yz;
253 R(2, 2) = 1 - mcth * (xx + yy);
254
255 }
