165inline void serialize(Archive & ar,
166 Eigen::Matrix<S, Rows_, Cols_, Ops_, MaxRows_, MaxCols_> & matrix,
167 ATTRIBUTE(maybe_unused) const unsigned int version)
168{
169 int rows = matrix.rows();
170 int cols = matrix.cols();
171 ar & make_nvp("rows", rows);
172 ar & make_nvp("cols", cols);
173 matrix.resize(rows, cols); // no-op if size does not change!
174
175 // always save/load row-major
176 for(int r = 0; r < rows; ++r)
177 for(int c = 0; c < cols; ++c)
178 ar & make_nvp("val", matrix(r,c));
179}
