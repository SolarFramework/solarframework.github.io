187inline void to_json(BasicJsonType& j, const Eigen::Matrix<bool, Dynamic, Dynamic, Ops_, MaxRows_, MaxCols_>& eigenData)

198template <class BasicJsonType, int Ops_, int MaxRows_, int MaxCols_>
199inline void from_json(const BasicJsonType& j, Eigen::Matrix<bool, Dynamic, Dynamic, Ops_, MaxRows_, MaxCols_>& eigenData)
200{
201 std::vector<int> data = j.template get<std::vector<int>>();
202 int size = (int)std::sqrt(data.size());
203 assert((size * size == data.size()) && "Must be a squared matrix");
204 eigenData.resize(size, size);
205 for (int x = 0; x < size; ++x)
206 for (int y = 0; y < size; ++y)
207 eigenData(x, y) = (bool)data[x * size + y];
208}

336inline bool saveToFile(const SolAR::datastructure::CameraRigParameters& camParams, std::string filePath)
337{
338 nlohmann::ordered_json j;
339 // write camera parameters to json
340 int nbCameras = camParams.cameraParams.size();
341 j["NbCameras"] = nbCameras;
342 int countCamera(0);
343 for (const auto& it : camParams.cameraParams) {
344 j["CameraParameters " + std::to_string(countCamera)] = it.second;
345 countCamera++;
346 }
347 // write rectifications to json
348 int nbRects = camParams.rectificationParams.size();
349 j["NbRectifications"] = nbRects;
350 int countRect(0);
351 for (const auto& it : camParams.rectificationParams) {
352 std::string nodeName = "Rectification " + std::to_string(countRect);
353 j[nodeName]["Id1"] = it.first.first;
354 j[nodeName]["Rect1"] = it.second.first;
355 j[nodeName]["Id2"] = it.first.second;
356 j[nodeName]["Rect2"] = it.second.second;
357 countRect++;
358 }
359 // write transformations to json
360 int nbTransformations = camParams.transformations.size();
361 j["NbTransformations"] = nbTransformations;
362 int countTrans(0);
363 for (const auto& it : camParams.transformations) {
364 std::string nodeName = "Transformation " + std::to_string(countTrans);
365 j[nodeName]["Id1"] = it.first.first;
366 j[nodeName]["Id2"] = it.first.second;
367 j[nodeName]["Transform"] = it.second;
368 countTrans++;
369 }
370 // save to file
371 std::ofstream f(filePath);
372 f << std::setw(4) << j;
373 f.close();
374 return true;
375}

377inline bool loadFromFile(SolAR::datastructure::CameraRigParameters& camParams, std::string filePath)
378{
379 std::ifstream f(filePath);
380 nlohmann::ordered_json j = nlohmann::ordered_json::parse(f, nullptr, false);
381 if (j.is_discarded()){
382 LOG_ERROR("Error when parsing file: {}", filePath);
383 return false;
384 }
385 if (!j.contains("NbCameras"))
386 return false;
387 // load camera parameters to json
388 int nbCameras = j["NbCameras"].get<int>();
389 for (int i = 0; i < nbCameras; ++i) {
390 CameraParameters cam = j["CameraParameters " + std::to_string(i)].get<CameraParameters>();
391 camParams.cameraParams[cam.id] = cam;
392 }
393 // load rectifications to json
394 if (j.contains("NbRectifications")) {
395 int nbRects = j["NbRectifications"].get<int>();
396 for (int i = 0; i < nbRects; ++i) {
397 std::string nodeName = "Rectification " + std::to_string(i);
398 uint32_t id1 = j[nodeName]["Id1"].get<uint32_t>();
399 uint32_t id2 = j[nodeName]["Id2"].get<uint32_t>();
400 RectificationParameters rect1 = j[nodeName]["Rect1"].get<RectificationParameters>();
401 RectificationParameters rect2 = j[nodeName]["Rect2"].get<RectificationParameters>();
402 camParams.rectificationParams[std::make_pair(id1, id2)] = std::make_pair(rect1, rect2);
403 }
404 }
405 // load transformations to json
406 if (j.contains("NbTransformations")) {
407 int nbTransformations = j["NbTransformations"].get<int>();
408 for (int i = 0; i < nbTransformations; ++i) {
409 std::string nodeName = "Transformation " + std::to_string(i);
410 uint32_t id1 = j[nodeName]["Id1"].get<uint32_t>();
411 uint32_t id2 = j[nodeName]["Id2"].get<uint32_t>();
412 Transform3Df transform = j[nodeName]["Transform"].get<Transform3Df>();
